Importando tus datos

Usa datos en formato GeoJSON

Página para generar GeoJSON data: http://geojson.io/#map=14/40.4762/-3.7157


var mapDiv2 = document.getElementById('map_canvas_2');
var latlng2 = new google.maps.LatLng(40.414933739888944,-3.683338165283203);
var myOptions2 ={
        center: latlng2,
        zoom: 10
        };
var map2 = new google.maps.Map(mapDiv2, myOptions2);
// NOTE: This uses cross-domain XHR, and may not work on older browsers.
map2.data.loadGeoJson('parques.geo.json');

Le podemos añadir un custom style a los datos:


var mapDiv3 = document.getElementById('map_canvas_3');
var myOptions3 ={
        center: latlng,
        zoom: 17
        };
var map3 = new google.maps.Map(mapDiv3, myOptions3);
// NOTE: This uses cross-domain XHR, and may not work on older browsers.
map3.data.loadGeoJson('paradigma.polygon.geo.json');

map3.data.setStyle({
  fillColor: '#E6152D',
  strokeWeight: 1
});
