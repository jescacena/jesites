    function initMap() {
        var mapDiv = document.getElementById('map');
        var map = new google.maps.Map(mapDiv, {
          center: {lat: 40.4314932, lng: -3.8106219},
          zoom: 17
        });
    }
                

    <script src="https://maps.googleapis.com/maps/api/js?callback=initMap" async defer></script>
